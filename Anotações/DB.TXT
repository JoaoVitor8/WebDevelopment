Criar uma tabela

create table nomeTabela(
	id_nomeTabela integer,
	nome varchar(40),
  email varchar(50),
  telefone integer,


	id_disciplina integer,
	primary key (id_nomeTabela),
	foreign key (id_disciplina)
	references disciplinas(id_disciplina)
)

---------------------------------------------------------------------------------------------------

Inserindo tabela

insert into nomeTabela values
